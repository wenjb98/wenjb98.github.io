一、栈的实现与基本操作

1、栈的链式存储结构

​ 栈的链式存储结构用双向链表加一个头指针、一个尾指针即可实现。用链式存储结构写栈的基本操作很简单即不在这附上具体代码，放一个结构体即可：

typedef struct{
    int e;
}SElemType;

typedef struct SNode
{
    SElemType data;
    struct SNode * prior;
    struct SNode * next;
}STNode,*pSNode;

typedef struct
{
    pSNode base;	//指向头结点
    pSNode top;		//指向为结点
}SqStack;

2、栈的顺序存储结构——基本操作

​ 严奶奶的书中具体介绍的是顺序存储结构为主，在这具体实现的代码便是栈的顺序存储结构。

//Stack.h
#define ok 1;
#define ERROR 0;
#define true 1;
#define false 0;

int STACK_INIT_SIZE = 100;
int STACKINCREMENT = 10;

typedef int Status;

typedef struct {
	char data;
}SElemType;

typedef struct {

	SElemType *base;
	SElemType *top;
	int stacksize;

}SqStack;

//栈基本操作
//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
Status InitStack(SqStack &S);			//构建栈

Status DestroyStack(SqStack &S);		//销毁栈

Status ClearStack(SqStack &S);			//若栈存在，将栈清空

Status StackEmpty(SqStack S);			//判断栈是否为空

int    StackLength(SqStack S);			//返回栈的长度

Status GetTop(SqStack S, SElemType &e); //查看栈顶元素

Status Push(SqStack &S, SElemType e);	//进栈

Status Pop(SqStack &S, SElemType &e);	//出栈

Status StackTraverse(SqStack S);		//遍历栈
//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+


Status InitStack(SqStack &S)			//构建栈
{
	S.base = (SElemType *)malloc(STACK_INIT_SIZE *sizeof(SElemType));
	if (!S.base) exit(0);
	S.top = S.base;
	S.stacksize = STACK_INIT_SIZE;
    return ok;
}

Status StackEmpty(SqStack S)//判断栈是否为空
{
	if (S.top == S.base) return ok;
	return ERROR;
}
Status DestroyStack(SqStack &S)			//销毁栈
{
	free(S.base);
	S.base = NULL;
	S.stacksize = 0;
	return ok;
}

Status ClearStack(SqStack &S)			//若栈存在，将栈清空
{
	if (StackEmpty(S)) return ok;
	while (!StackEmpty(S))
		--S.top;
	//S.stacksize = 0;
	return ok;
}

int    StackLength(SqStack S)			//返回栈的长度
{
	int i = 0;
	i = S.top - S.base;
	/*while (p != S.top)
	{
		++p;
		++i;
	}*/
	return i;
}
Status GetTop(SqStack S, SElemType &e) //查看栈顶元素
{
	if (StackEmpty(S)) return ERROR;
	SElemType *p;
	p = S.top;
	--p;
	e = *p;
	return ok;
}

Status Push(SqStack &S, SElemType e) //进栈
{
	if (S.top - S.base >= S.stacksize)
	{
		S.base = (SElemType *)realloc(S.base,(S.stacksize+ STACKINCREMENT)*sizeof(SElemType));
		if (!S.base) exit(0);
		S.top = S.base+S.stacksize;
		S.stacksize = S.stacksize + STACKINCREMENT;
	}
	*S.top = e;
	++S.top;
	return ok;
}

Status Pop(SqStack &S, SElemType &e) //出栈
{
	if (StackEmpty(S)) return ERROR;
	--S.top;
	e = *S.top;
	return ok;
}
Status StackTraverse(SqStack S)
{
	SElemType *p = S.base;
	if (NULL == S.base)
		return ERROR;
	while (p < S.top)
	{
		printf("%d", p->data);
		++p;
	}
	
	printf("\n");
	return ok;
}

	SqStack S;

	int N = e.data;
	
	while (N)
	{
		e.data = N%2;
		Push(S, e);
		N = N / 2;
	}
	while (!StackEmpty(S))
	{
		Pop(S, e);

	}
	return;

typedef struct {

//括号匹配监测
//()、[]
Status matching(char *a, int n);
Status matching(char *a, int n)
{
	int i = 1;
	SqStack S;
	InitStack(S);
	SElemType e;
	SElemType ea;
	while (i <= n)//n 为数组长度  i为数组下标
	{
		switch (a[i])
		{
		case '(': 
			e.data = a[i];
			Push(S, e);
			i++;
			break;
		case '[':
			e.data = a[i];
			Push(S, e);
			i++;
			break;
		case ')':
			 GetTop(S, ea);
			if (ea.data == '(')
			{
				Pop(S,ea);
				i++;
			}
			else i++;
			break;
		case ']':
			 GetTop(S, ea);
			if (ea.data == '[')
			{
				Pop(S, ea);
				i++;
			}
			else i++;
			break;
		}
	}
	if (StackEmpty(S)) return ok;
	return ERROR;
}

3.4、迷宫求解
typedef struct {
	int i;
	int j;
}PosType;

typedef struct {
	int ord;		//序号
	PosType seat;	//坐标位置
	int di;			//方向
}SElemType;

/*
	-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
	迷宫求解-需要的所有函数.

	0:路，可通过
	1:墙，不可通过;且同时作为不能通过的标记
	2:沿途的标记

	-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
*/
Status Pass(PosType e);

void FootPrint(PosType e);

PosType NextPos(PosType e, int N);

void MarkPrint(PosType e);

Status MazePath(PosType start, PosType end);

//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

Status Pass(PosType e)
{
	int N = carry[e.i][e.j];
	if (!N) return true;
	return false;
}
void FootPrint(PosType e)
{
	carry[e.i][e.j] = 2;
	return;
}
PosType NextPos(PosType e, int N)
{
	switch (N)
	{
	case 1:
		e.j = e.j + 1;
		break;
	case 2:
		e.i = e.i + 1;
		break;
	case 3:
		e.j = e.j - 1;
		break;
	case 4:
		e.i = e.i - 1;
		break;
	}
	return e;
}
void MarkPrint(PosType e)
{
	carry[e.i][e.j] = 1;
	return;
}


Status MazePath(PosType start, PosType end)
{
	SqStack S;
	InitStack(S);
	SElemType e;
	PosType curpos;
	curpos = start;		//设置“当前位置”为“入口位置”
	
	int curstep = 1;	//步数
	do {
		if (Pass(curpos)) {
			FootPrint(curpos);	//标记
			e.ord = curstep;
			e.seat = curpos;
			e.di = 1;
			Push(S, e);
			if ((curpos.i == end.i) && (curpos.j == end.j)) { StackTraverse(S); return true; }
			curpos = NextPos(curpos, 1);
			curstep++;
		}
		else
		{
			if (!StackEmpty(S))
			{
				Pop(S,e);
				while (e.di == 4 && !StackEmpty(S))
				{
					MarkPrint(e.seat);
					Pop(S, e);
				}//while
				if (e.di < 4)
				{
					e.di++;
					Push(S, e);
					curpos = NextPos(e.seat, e.di);

				}//if
			}//if

		}//else

	} while (!StackEmpty(S));
	return false;
}
3.5、表达式求值

​ 严奶奶书中的原程序，输入的是一段字符串，加减一直ASCii值相加减，我想不出来，加减的操作数应该是整形，操作符号应该是字符型，肯定不能放在相同类型的栈里面。

​ 自己想出来的方法是在定义一个int类型数组(这个数组模拟栈的先进后出模式)将操作数放在整形数组里面，将操作符放在原先定义的栈里面。

typedef struct {
	char data;
}SElemType;


//表达式求值
typedef int OPerandType;

OPerandType OPND[50];
OPerandType iod=0;

//函数
Status In(char c);

char Precede(SElemType e, char c);

OPerandType Operate(OPerandType a, char theta, OPerandType b);

OPerandType EvaluateExpression();

int Num(char *b, int i);//数值转换

//具体代码实现
Status In(char c)  //判断是否是数字
{
	switch (c)
	{
	case '+':
	case '-':
	case '*':
	case '/':
	case '(':
	case ')':
	case '#':
		return true;
	default:
		return false;
	}
}

char Precede(SElemType e, char c)
{
	char xita[] = { '+','-','*','/','(',')','#' };
	char sita[7][7] =
	{
		{ '>','>','<','<','<','>','>' },
		{ '>','>','<','<','<','>','>' },
		{ '>','>','>','>','<','>','>' },
		{ '>','>','>','>','<','>','>' },
		{ '<','<','<','<','<','=','0' },
		{ '>','>','>','>','0','>','>' },
		{ '<','<','<','<','<','0','=' }
	};

	int i, j;
	for(int p=0;p<7;p++)
		if (xita[p] == e.data)
		{
			i = p;
			break;
		}
	for (int p = 0; p < 7; p++)
		if (xita[p] == c)
		{
			j = p;
			break;
		}
	return sita[i][j];
}

OPerandType Operate(OPerandType a, char theta, OPerandType b)
{
	OPerandType sum = 0;
	switch (theta)
	{
	case '+':sum = a + b; break;
	case '-':sum = a - b; break;
	case '*':sum = a * b; break;
	case '/':sum = a / b; break;
	}
	return sum;
}

int Num(char *b, int i)//数值转换
{
	int p = 0;
	int sum = 0;
	int s = 1;
	for (p = 1; p < i; p++)
		s = s * 10;
	for (p = 0; p < i; p++)
	{
		sum = (b[p] - '0') * s + sum;
		s = s / 10;
	}
	return sum;
}

OPerandType EvaluateExpression()
{
	char charNum[100]; 
	int ic = 0;
	SqStack OPTR; 
	InitStack(OPTR);
	SElemType e, d, x, e1, theta;
	int a, b;
	e.data = '#';
	Push(OPTR, e);
	printf(">");
	char c = getchar();
	GetTop(OPTR, e1);
	while (c != '#' ||  e1.data!= '#')
	{
		if (!In(c))
		{
			charNum[ic] = c;
			++ic;
			c = getchar();
		}
		else
		{
			if (ic != 0)
			{
				OPND[iod] = Num(charNum, ic);
				++iod;
				ic = 0;
			}
			GetTop(OPTR, e1);
			switch (Precede(e1, c))
			{
			case '<':
				e.data = c;
				Push(OPTR, e);
				c = getchar();
				break;
			case '=':
				Pop(OPTR, x);
				c = getchar(); 
				break;
			case '>':
				Pop(OPTR, theta);
				b = OPND[--iod];
				a = OPND[--iod];
				OPND[iod] = Operate(a, theta.data, b);
				++iod;
				break;
			}//switch
			GetTop(OPTR, e1);
		}//while
	}
	return OPND[--iod];
}

二、队的实现与基本操作

1、队的链式存储结构

#pragma once
#define ok 1;
#define ERROR 0;
#define true 1;
#define false 0;
//#define MAXSIZE 100;
int MAXSIZE = 10;
#define OVERFLOW 0
typedef struct {
	int e;
}QueueElemType;

typedef struct {
	QueueElemType *base;
	int front;
	int rear;
}SqQueue;


//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

Status InitQueue(SqQueue &Q);

Status DestroyQueue(SqQueue &Q);

Status QueueEmpty(SqQueue &Q);

int QueueLenth(SqQueue &Q);

Status GEtaHead(SqQueue &Q, QueueElemType &e);

Status EnQueue(SqQueue &Q, QueueElemType e);	//插入元素->进队

Status DeQueue(SqQueue &Q, QueueElemType &e);	//删除元素->出队
	
Status QueueTraverse(SqQueue Q);

Status InitQueue(SqQueue &Q);

//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

Status InitQueue(SqQueue &Q)
{
	Q.base = (QueueElemType  *)malloc(sizeof(QueueElemType)*MAXSIZE);
	if (!Q.base) exit(OVERFLOW);
	Q.front = 0;
	Q.rear = 0;
	return ok;
}

int QueueLenth(SqQueue &Q)
{
	return (Q.rear - Q.front + MAXSIZE) % MAXSIZE;
}

Status EnQueue(SqQueue &Q, QueueElemType e)
{
	if ((Q.rear + 1) % MAXSIZE == Q.front) return ERROR;
	Q.base[Q.rear] = e;
	Q.rear = (Q.rear + 1) % MAXSIZE;
	return ok;
}

Status DeQueue(SqQueue &Q, QueueElemType &e)
{
	if (Q.front == Q.rear) return ERROR;
	e = Q.base[Q.front];
	Q.front = (Q.front + 1) % MAXSIZE;
	return ok;
}

Status DestroyQueue(SqQueue &Q)
{
	if (!Q.base)
		return ok;
	free(Q.base);
	Q.base = NULL;
	Q.front = 0;
	Q.rear = 0;
	return ok;
}

Status QueueEmpty(SqQueue &Q)
{
	if (Q.front == Q.rear) return ok;
	return ERROR;
}

Status GEtaHead(SqQueue &Q, QueueElemType &e)
{
	if (QueueEmpty(Q)) return ERROR;
	e = Q.base[Q.front];
	return ok;
}

Status QueueTraverse(SqQueue Q)
{
	int i = Q.front;
	while (i != Q.rear)
	{
		printf("%d", Q.base[i]);
		i = (i + 1) % MAXSIZE;
	}
	printf("\n");
	return 0;
}

2、队的顺序存储——循环队列

  • 求队长的公式：(Q.rear-Q.front+MAXSIZE)%MAXSIZE

  • 队尾插入数据队满条件：(Q.rear+1)%MAXSIZE==Q.front

  • 对头对尾下标加一：Q.front=(Q.front+1)%MAXSIZE

#pragma once
#include "Stack.h"

#define OVERFLOW 0

typedef struct {
	int e;
}QElemType;

typedef struct QNode {
	QElemType data;
	struct QNode *next;

}QNode,*pQueuePtr;

typedef struct {

	pQueuePtr front;	//队头指针

	pQueuePtr rear;		//队尾指针

}LinkQueue;


//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+


Status InitQueue(LinkQueue &Q); 

Status DestroyQueue(LinkQueue &Q);

Status QueueEmpty(LinkQueue &Q);

int QueueLenth(LinkQueue &Q);

Status GEtaHead(LinkQueue &Q, QElemType &e);

Status EnQueue(LinkQueue &Q, QElemType e);

Status DeQueue(LinkQueue &Q, QElemType &e);

Status QueueTraverse(LinkQueue Q);


//-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

Status InitQueue(LinkQueue &Q)
{
	Q.front = Q.rear = (pQueuePtr)malloc(sizeof(QNode));
	if (!Q.front) exit(OVERFLOW);
	Q.front->next = NULL;
	return ok;
}

Status DestroyQueue(LinkQueue &Q)
{
	while (Q.front)
	{
		Q.rear = Q.front->next;
		free(Q.front);
		Q.front = Q.rear;

	}
	return ok;
}

Status EnQueue(LinkQueue &Q, QElemType e)
{
	pQueuePtr p;
	p = (pQueuePtr)malloc(sizeof(QNode));
	if (!p) exit(OVERFLOW);
	p->data = e;
	p->next = NULL;
	Q.rear->next = p;
	Q.rear = p;
	return ok;
}

Status DeQueue(LinkQueue &Q, QElemType &e)
{
	if (Q.front == Q.rear) return ERROR;
	pQueuePtr p;
	p = Q.front->next;
	e = p->data;
	Q.front->next = p->next;
	free(p);
	return ok;
}

Status QueueEmpty(LinkQueue &Q)
{
	if (Q.front == Q.rear) return ok;
	return ERROR;
}

Status GEtaHead(LinkQueue &Q, QElemType &e)
{
	if (QueueEmpty(Q)) return ERROR;
	e = Q.front->next->data;
	return ok;
}

int QueueLenth(LinkQueue &Q)
{
	int i = 0;
	pQueuePtr p;
	p = Q.front;
	if (p->next)
	{
		++i;
		p = p->next;
	}
	return i;
}

Status QueueTraverse(LinkQueue Q)
{
	pQueuePtr p;
	p = Q.front->next;
	while (p)
	{
		printf("%d", p->data);
		p = p->next;
	}
	printf("\n");
	return ok;
}
